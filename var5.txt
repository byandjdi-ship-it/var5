// --- ГЕНЕРАЦИЯ ТЕСТА ---
const container = document.getElementById('list');
let html = '';
let answeredState = {}; // храним состояние каждого вопроса

questions.forEach((q, qIndex) => {
    html += `<div class="q" id="q${qIndex}">`;
    html += `<div class="text"><span class="num">${q.id}</span> ${q.text}</div>`;
    html += `<div class="opts">`;
    
    q.options.forEach((opt, optIndex) => {
        // Определяем цвет: green если правильный, red если нет
        const color = (optIndex === q.correct) ? 'green' : 'red';
        html += `<div class="opt" data-q="${qIndex}" data-opt="${optIndex}" data-color="${color}">${opt}</div>`;
    });
    
    html += `</div></div>`;
});

container.innerHTML = html;

// --- ЛОГИКА (МАКСИМАЛЬНО ПРОСТАЯ!) ---
let correctCount = 0;
let wrongCount = 0;
let answeredCount = 0;
const answeredSet = new Set();

document.querySelectorAll('.opt').forEach(el => {
    el.addEventListener('click', function() {
        const qIndex = parseInt(this.dataset.q);
        const optIndex = parseInt(this.dataset.opt);
        const key = `${qIndex}-${optIndex}`;
        
        // Если уже отвечали на этот вопрос — игнорируем
        if (answeredSet.has(qIndex)) return;
        answeredSet.add(qIndex);
        
        // Применяем цвет из data-color (САМАЯ ГЛАВНАЯ СТРОКА!)
        const color = this.dataset.color;
        this.classList.add(color);
        
        // Дополнительно подсвечиваем правильный ответ (для наглядности)
        const question = questions[qIndex];
        const correctOptIndex = question.correct;
        const allOpts = document.querySelectorAll(`#q${qIndex} .opt`);
        allOpts.forEach(opt => {
            const idx = parseInt(opt.dataset.opt);
            if (idx === correctOptIndex) {
                opt.classList.add('green');
            }
            opt.classList.add('disabled');
        });
        
        // Обновляем статистику
        if (optIndex === correctOptIndex) {
            correctCount++;
        } else {
            wrongCount++;
        }
        answeredCount++;
        
        document.getElementById('cCount').textContent = correctCount;
        document.getElementById('wCount').textContent = wrongCount;
        document.getElementById('aCount').textContent = answeredCount;
    });
});

// --- СБРОС ---
document.getElementById('resetBtn').addEventListener('click', function() {
    document.querySelectorAll('.opt').forEach(el => {
        el.classList.remove('green', 'red', 'disabled');
    });
    answeredSet.clear();
    correctCount = 0;
    wrongCount = 0;
    answeredCount = 0;
    document.getElementById('cCount').textContent = '0';
    document.getElementById('wCount').textContent = '0';
    document.getElementById('aCount').textContent = '0';
});
